# RREF demo: explore row-reduction, rank, and null space
# Define a 3x4 matrix with dependent rows
A = [
  [1, 2,  -1,  3],
  [2, 4,  -2,  6],
  [1, 1,   0,  1]
]

# Show RREF and rank
rref(A)
rank(A)

# Null space basis (of columns)
nullspace(A)

# Try a slightly perturbed version to see rank change
B = [
  [1, 2,  -1,  3],
  [2, 4.1,-2,  6],
  [1, 1,   0,  1]
]
rank(B)
rref(B)
